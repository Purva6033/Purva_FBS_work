#include <stdio.h>

int isPrime(int num) {
    if (num <= 1)
        return 0;
    for (int i = 2; i <= num / 2; i++) {
        if (num % i == 0)
            return 0;
    }
    return 1;
}

int main() {
    int start, end, sum = 0;

    printf("Enter range: ");
    scanf("%d %d", &start, &end);

    for (int i = start; i <= end; i++) {
        if (isPrime(i)) {
            sum += i;
        }
    }

    printf("Sum of prime numbers = %d", sum);

    return 0;
}








#include <stdio.h>

int main() {
    char str[100], ch;

    printf("Enter string: ");
    scanf("%s", str);

    printf("Enter character to replace: ");
    scanf(" %c", &ch);

    for (int i = 0; str[i] != '\0'; i++) {
        if (str[i] == ch) {
            str[i] = '$';
        }
    }

    printf("Updated string: %s", str);

    return 0;
}





#include <stdio.h>
#include <string.h>

struct Task {
    int id;
    char desc[100];
    char status[20];
};

struct Task t[100];
int count = 0;

void addTask() {
    printf("Enter Task ID: ");
    scanf("%d", &t[count].id);

    printf("Enter Description: ");
    scanf(" %[^\n]", t[count].desc);

    strcpy(t[count].status, "Pending");

    count++;
}

void updateTask() {
    int id, found = 0;
    printf("Enter Task ID to update: ");
    scanf("%d", &id);

    for (int i = 0; i < count; i++) {
        if (t[i].id == id) {
            printf("Enter new status: ");
            scanf("%s", t[i].status);
            found = 1;
        }
    }

    if (!found)
        printf("Task not found!\n");
}

void displayTasks() {
    for (int i = 0; i < count; i++) {
        printf("\nID: %d", t[i].id);
        printf("\nDescription: %s", t[i].desc);
        printf("\nStatus: %s\n", t[i].status);
    }
}

int main() {
    int choice;

    do {
        printf("\n1. Add Task");
        printf("\n2. Update Task");
        printf("\n3. Display Tasks");
        printf("\n4. Exit");
        printf("\nEnter choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1: addTask(); break;
            case 2: updateTask(); break;
            case 3: displayTasks(); break;
            case 4: printf("Exiting..."); break;
            default: printf("Invalid choice!");
        }

    } while (choice != 4);

    return 0;
}

};

int count = 0;

void addTask() {
    printf("Enter Task ID: ");
    scanf("%d", &t[count].id);

    printf("Enter Description (one word): ");
    scanf("%s", t[count].desc);

    printf("Enter Status (Pending/Done): ");
    scanf("%s", t[count].status);

    count++;
}

void updateTask() {
    int id, i;

    printf("Enter Task ID to update: ");
    scanf("%d", &id);

    for (i = 0; i < count; i++) {
        if (t[i].id == id) {
            printf("Enter new status: ");
            scanf("%s", t[i].status);
            return;
        }
    }

    printf("Task not found!\n");
}

void displayTasks() {
    int i;

    for (i = 0; i < count; i++) {
        printf("\nID: %d", t[i].id);
        printf("\nDescription: %s", t[i].desc);
        printf("\nStatus: %s\n", t[i].status);
    }
}

int main() {
    int choice;

    do {
        printf("\n1. Add Task");
        printf("\n2. Update Task");
        printf("\n3. Display Tasks");
        printf("\n4. Exit");

        printf("\nEnter choice: ");
